001package com.github.dylon.liblevenshtein.serialization;
002
003import java.io.InputStream;
004import java.io.OutputStream;
005import java.io.Serializable;
006
007/**
008 * (De)Serialization utilities for various interchange formats.
009 */
010public interface Serializer {
011
012  /**
013   * Serializes an object to some output stream.
014   * @param object Object to serialize.
015   * @param stream Stream into which to write the serialization.
016   * @throws Exception When the object cannot be serialized
017   * (implementation-specific).
018   */
019  void serialize(Serializable object, OutputStream stream) throws Exception;
020
021  /**
022   * Serializes an object and returns its bytes.
023   * @param object Object to serialize.
024   * @return Bytes of the serialized object.
025   * @throws Exception When the object cannot be serialized
026   * (implementation-specific).
027   */
028  byte[] serialize(Serializable object) throws Exception;
029
030  /**
031   * Deserializes an object from some stream.
032   * @param type Class of the deserialized object.
033   * @param stream Stream from which to deserialize the object.
034   * @param <Type> Type of the deserialized object.
035   * @return Deserialized object from the stream.
036   * @throws Exception When the object cannot be deserialized
037   * (implementation-specific).
038   */
039  <Type> Type deserialize(Class<Type> type, InputStream stream) throws Exception;
040
041  /**
042   * Deserializes an object from its byes.
043   * @param type Class of the deserialized object.
044   * @param bytes Bytes of the serialized object.
045   * @param <Type> Type of the deserialized object.
046   * @return Deserialized object from its bytes.
047   * @throws Exception When the object cannot be deserialized
048   * (implementation-specific).
049   */
050  <Type> Type deserialize(Class<Type> type, byte[] bytes) throws Exception;
051}
