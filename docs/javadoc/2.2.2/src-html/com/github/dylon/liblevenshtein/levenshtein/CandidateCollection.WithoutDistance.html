001package com.github.dylon.liblevenshtein.levenshtein;
002
003import java.io.Serializable;
004import java.util.ArrayList;
005import java.util.Collection;
006import java.util.Iterator;
007
008import org.apache.commons.lang3.builder.ReflectionToStringBuilder;
009import org.apache.commons.lang3.builder.ToStringStyle;
010
011/**
012 * <p>
013 * <b>WARNING</b> This class is deprecated and will be removed in the next major
014 * release.  Please migrate your code to use
015 * {@link com.github.dylon.liblevenshtein.levenshtein.factory.ICandidateFactory}
016 * and an instance of
017 * {@link com.github.dylon.liblevenshtein.levenshtein.ICandidateCollection}
018 * that can make use of it.
019 * </p>
020 *
021 * <p>
022 * Collection of all candidates from the dictionary that are no further than the
023 * specified distance from the query term.
024 * </p>
025 *
026 * @param <Type> Kind of the spelling candidates returned from the dictionary.
027 *
028 * @author Dylon Edwards
029 * @since 2.1.0
030 */
031@Deprecated
032public abstract class CandidateCollection<Type>
033  implements ICandidateCollection<Type>, Serializable {
034
035  private static final long serialVersionUID = 1L;
036
037  /**
038   * Maximum number of correction candidates that may be added to this
039   * ICandidateCollection.
040   */
041  protected final int maxCandidates;
042
043  /**
044   * Correction candidates that have been added to this ICandidateCollection.
045   */
046  protected final Collection<Type> candidates;
047
048  /**
049   * Constructs a new instance of ICandidateCollection, with a maximum threshold
050   * for the number of correction candidates it will accept.
051   * @param maxCandidates Maximum number of correction candidates that may be
052   * added to this collection.
053   */
054  public CandidateCollection(final int maxCandidates) {
055    this.maxCandidates = maxCandidates;
056    // TODO: Consider specializing the data struction by <Type>, such as using
057    // an unsorted Dawg for <Type=String>, etc.
058    this.candidates = (maxCandidates < Integer.MAX_VALUE)
059      ? new ArrayList<>(maxCandidates)
060      : new ArrayList<>();
061  }
062
063  /**
064   * {@inheritDoc}
065   */
066  @Override
067  public Iterator<Type> iterator() {
068    return candidates.iterator();
069  }
070
071  /**
072   * {@inheritDoc}
073   */
074  @Override
075  public String toString() {
076    return ReflectionToStringBuilder.toString(this, ToStringStyle.SHORT_PREFIX_STYLE);
077  }
078
079  /**
080   * Implementation of {@link CandidateCollection} that includes the distances
081   * with the spelling candidates, packaged in {@link Candidate}s.
082   * @author Dylon Edwards
083   * @since 2.1.0
084   */
085  public static class WithDistance extends CandidateCollection<Candidate> {
086
087    private static final long serialVersionUID = 1L;
088
089    /**
090     * Constructs a new instance of ICandidateCollection, with a maximum
091     * threshold for the number of correction candidates it will accept.
092     * @param maxCandidates Maximum number of correction candidates that may be
093     * added to this collection.
094     */
095    public WithDistance(final int maxCandidates) {
096      super(maxCandidates);
097    }
098
099    /**
100     * {@inheritDoc}
101     */
102    @Override
103    public boolean offer(final String term, final int distance) {
104      if (candidates.size() == maxCandidates) {
105        return false;
106      }
107
108      candidates.add(new Candidate(term, distance));
109      return true;
110    }
111  }
112
113  /**
114   * Implementation of {@link CandidateCollection} that returns only the raw,
115   * spelling candidates.
116   * @author Dylon Edwards
117   * @since 2.1.0
118   */
119  public static class WithoutDistance extends CandidateCollection<String> {
120
121    private static final long serialVersionUID = 1L;
122
123    /**
124     * Constructs a new instance of ICandidateCollection, with a maximum
125     * threshold for the number of correction candidates it will accept.
126     * @param maxCandidates Maximum number of correction candidates that may be
127     * added to this collection.
128     */
129    public WithoutDistance(final int maxCandidates) {
130      super(maxCandidates);
131    }
132
133    /**
134     * {@inheritDoc}
135     */
136    @Override
137    public boolean offer(final String term, final int distance) {
138      if (candidates.size() == maxCandidates) {
139        return false;
140      }
141
142      candidates.add(term); // NOTE: distance is ignored ...
143      return true;
144    }
145  }
146}
