001package com.github.liblevenshtein.transducer.factory;
002
003import java.io.Serializable;
004
005import com.github.liblevenshtein.transducer.IState;
006
007/**
008 * Builds Levenshtein states.
009 * @author Dylon Edwards
010 * @since 2.1.0
011 */
012public interface IStateFactory extends Serializable {
013
014  long serialVersionUID = 1L;
015
016  /**
017   * Builds a new, Levenshtein state with the given position vectors.
018   * @param positions Array of position vectors to link into the state.
019   * @return New state having the position vectors.
020   */
021  IState build(int[]... positions);
022
023  /**
024   * Recycles a state for re-use.
025   * @param state {@link IState} to recycle.  You must discard the state once
026   * you've recycled it.
027   */
028  default void recycle(final IState state) {
029    // default behavior is to do nothing
030  }
031}
